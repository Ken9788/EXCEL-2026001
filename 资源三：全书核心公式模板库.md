\*\*查找类\*\*



\\=VLOOKUP(查找值, 查找区域, 返回列号, 0)



\\=XLOOKUP(查找值, 查找列, 返回列, "缺省值")



\\=INDEX(返回列, MATCH(查找值, 查找列, 0))



\*\*统计类\*\*



\\=SUMIFS(求和列, 条件列1, 条件1, 条件列2, 条件2)



\\=COUNTIFS(条件列1, 条件1, 条件列2, 条件2)



\\=AVERAGEIFS(平均列, 条件列1, 条件1, 条件列2, 条件2)



\\=SUBTOTAL(9, 区域)



\*\*逻辑类\*\*



\\=IF(条件, "成立显示", "不成立显示")



\\=IFS(条件1, "结果1", 条件2, "结果2", TRUE, "默认")



\\=IFERROR(原公式, "出错显示")



\*\*文本类\*\*



\\=LEFT(文本, 字符数)



\\=RIGHT(文本, 字符数)



\\=MID(文本, 起始位置, 字符数)



\\=FIND("找什么", 文本)



\\=LEN(文本)



\\=TRIM(文本)



\\=文本1 \& "分隔符" \& 文本2



\*\*日期类\*\*



\\=YEAR(日期)



\\=MONTH(日期)



\\=DAY(日期)



\\=WEEKDAY(日期, 2)



\\=DATEDIF(开始, 结束, "D")
